50. Print positions and values of elements in an array < 5

Write a C program to read an array of length 5 and print the position and value of the array elements of value less than 5.
Test Data:
Input the 5 members of the array:
15
25
4
35
40
Expected Output:
A[2] = 4.0


```c
#include <stdio.h>
#define AL 5
int main() {
    // Define the size of the array
    int array_n[AL], i, temp;
	
    // Prompt user to input 5 elements of the array
    printf("Input the 5 members of the array:\n");
    for(i = 0; i <  AL; i++) {
        scanf("%d", &array_n[i]);
    }

    // Swap the first half of the array with the second half
    for(i = 0; i < AL; i++) {
        if(i < (AL/2)) {
            temp = array_n[i];
            array_n[i] = array_n[AL-(i+1)];
            array_n[AL-(i+1)] = temp;
        }
    }

    // Print the modified array
    for(i = 0; i < AL; i++) {
        printf("array_n[%d] = %d\n", i, array_n[i]);
    }

    return 0;
}
```
